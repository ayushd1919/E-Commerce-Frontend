Pattern — all services use providedIn: 'root'
// every service follows this pattern — no module needed @Injectable({ providedIn: 'root' }) export class AuthService { private currentUser$ = new BehaviorSubject<User | null>(null); readonly user$ = this.currentUser$.asObservable(); isLoggedIn() { return !!this.currentUser$.getValue(); } getRole() { return this.currentUser$.getValue()?.role; } }

Standalone apps use functional guards — no classes needed
// auth.guard.ts export const authGuard: CanActivateFn = () => { const auth = inject(AuthService); const router = inject(Router); return auth.isLoggedIn() ? true : router.createUrlTree(['/auth/login']); }; // admin.guard.ts export const adminGuard: CanActivateFn = () => { const auth = inject(AuthService); const router = inject(Router); return auth.getRole() === 'admin' ? true : router.createUrlTree(['/']); }; // guest.guard.ts — stops logged-in users hitting /auth/login export const guestGuard: CanActivateFn = () => { const auth = inject(AuthService); const router = inject(Router); return auth.isLoggedIn() ? router.createUrlTree(['/']) : true; }; // admin-load.guard.ts — CanMatchFn stops bundle download for non-admins export const adminLoadGuard: CanMatchFn = () => { return inject(AuthService).getRole() === 'admin'; };

Standalone apps use functional interceptors (HttpInterceptorFn)
// credentials.interceptor.ts — must be FIRST in the chain export const credentialsInterceptor: HttpInterceptorFn = (req, next) => next(req.clone({ withCredentials: true })); // loading.interceptor.ts export const loadingInterceptor: HttpInterceptorFn = (req, next) => { const loading = inject(LoadingService); loading.show(); return next(req).pipe(finalize(() => loading.hide())); }; // error.interceptor.ts export const errorInterceptor: HttpInterceptorFn = (req, next) => { const auth = inject(AuthService); const router = inject(Router); return next(req).pipe( catchError((err: HttpErrorResponse) => { if (err.status === 401) { auth.clearSession(); router.navigate(['/auth/login']); } if (err.status === 403 && err.error?.reason === 'locked') { auth.clearSession(); router.navigate(['/auth/login'], { queryParams: { reason: 'locked' }}); } return throwError(() => err); }) ); };
Every component has standalone: true and imports its own dependencies

navbar.component shared

  • imports: RouterLink, AsyncPipe, NgIf
  • Injects: AuthService, CartService
  • Shows guest / customer / admin nav items via *ngIf
  • Cart badge subscribes to cart.itemCount$
  • Search bar emits to router query param ?q=

product-card.component shared

  • imports: RouterLink, NgIf, ImageFallbackDirective
  • @Input() product: Product
  • Shows image with fallback directive on error
  • Price formatted with CurrencyFormatPipe
  • No add-to-cart logic — handled on detail page

filter-panel.component shared

  • imports: ReactiveFormsModule, NgFor, AsyncPipe
  • Injects: TaxonomyService
  • Cascading dropdowns: Type → Category → SubCat
  • Price min/max + in-stock toggle
  • @Output() filterChange EventEmitter

image-fallback.directive shared

  • Attribute directive on <img> tags
  • Listens to (error) event
  • Sets src to assets/images/placeholder.png
  • Prevents broken image icons across the whole app

product-detail.component

  • imports: NgIf, RouterLink, ImageFallbackDirective
  • Route input: @Input() id: string (via withComponentInputBinding)
  • Breadcrumb: Type → Category → SubCat → Product
  • Add to Cart button: shown only if isLoggedIn()
  • Share button: navigator.clipboard.writeText(location.href)

forgot-password.component

  • 3-step stepper managed with local step: 1|2|3 signal
  • Step 1: email → calls forgotPassword() → shows code on screen
  • Step 2: code entry → calls verifyCode()
  • Step 3: new password form → calls resetPassword()
  • Shows expiry error if code is stale

checkout.component

  • imports: ReactiveFormsModule, NgFor, CurrencyFormatPipe
  • Injects: CartService, OrderService, Router
  • Payment radio: Credit Card / Debit Card / COD / Bank Transfer
  • On confirm → orderService.placeOrder() → cartService.clearCart() → navigate to /checkout/confirmation/:id

admin-product-form.component

  • Same component for create and edit — detect via route param :id
  • imports: ReactiveFormsModule, NgFor, NgIf
  • Cascading dropdowns via TaxonomyService
  • File input → preview → send as FormData
  • Patches form values from product when editing

admin-layout.component

  • imports: RouterOutlet, RouterLink, RouterLinkActive
  • Sidebar with links: Products / Customers / Orders
  • Wraps all admin child routes via nested <router-outlet>
  • No navbar shown inside admin — self-contained layout

admin-customer-list.component

  • imports: NgFor, NgIf, AsyncPipe
  • Injects: UserService
  • Lock/Unlock toggle per row
  • Optimistic UI update — reflects change immediately
  • Lock enforced server-side on customer's next request
